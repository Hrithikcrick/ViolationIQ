$ErrorActionPreference = "Stop"

# 1. Create reports folder
New-Item -ItemType Directory -Force -Path "reports" | Out-Null

# 2. Save LaTeX report from clipboard
# First copy your full Overleaf LaTeX code, then run this script
$latex = Get-Clipboard -Raw

if ($latex -notmatch "\\documentclass") {
    Write-Host "Your clipboard does not contain LaTeX code. Copy the full Overleaf main.tex code first, then run again." -ForegroundColor Red
    exit
}

$texPath = "reports\ViolationIQ_IEEE_Report.tex"
Set-Content -Path $texPath -Value $latex -Encoding UTF8

Write-Host "Saved LaTeX report:" $texPath -ForegroundColor Green

# 3. Copy latest PDF from Downloads if available
# Make sure the latest PDF in Downloads is your Overleaf compiled report
$latestPdf = Get-ChildItem "$env:USERPROFILE\Downloads" -Filter "*.pdf" -ErrorAction SilentlyContinue |
    Sort-Object LastWriteTime -Descending |
    Select-Object -First 1

if ($latestPdf) {
    $pdfPath = "reports\ViolationIQ_IEEE_Report.pdf"
    Copy-Item $latestPdf.FullName $pdfPath -Force
    Write-Host "Copied latest PDF to:" $pdfPath -ForegroundColor Green
    Write-Host "Source PDF was:" $latestPdf.FullName -ForegroundColor Yellow
} else {
    Write-Host "No PDF found in Downloads. Only .tex report will be added." -ForegroundColor Yellow
}

# 4. Add a small report README
$reportReadme = @'
# ViolationIQ Project Report

This folder contains the IEEE-style project report for ViolationIQ.

## Files

| File | Description |
|---|---|
| `ViolationIQ_IEEE_Report.tex` | Overleaf / LaTeX source code of the report |
| `ViolationIQ_IEEE_Report.pdf` | Compiled project report PDF, if available |

## Report Coverage

The report explains:

- problem statement,
- proposed approach,
- adaptive multi-expert architecture,
- datasets used,
- model training results,
- helmet and rider evidence module,
- number plate detection and safe OCR,
- red-light video evidence,
- signboard context evidence,
- evidence reasoning and safety layer,
- repository structure,
- deployment workflow,
- limitations and future scope.

ViolationIQ is presented as an AI evidence copilot for traffic enforcement, not a blind automatic challan generator.
'@

Set-Content -Path "reports\README.md" -Value $reportReadme -Encoding UTF8

# 5. Add report section in main README
$mainReadme = "README.md"

if (Test-Path $mainReadme) {
    $readme = Get-Content $mainReadme -Raw

    $reportSection = @'
<!-- PROJECT_REPORT_START -->

---

## 📄 Project Report

The complete IEEE-style project report is available in the `reports/` folder.

| File | Link |
|---|---|
| LaTeX Source | [`reports/ViolationIQ_IEEE_Report.tex`](reports/ViolationIQ_IEEE_Report.tex) |
| Compiled PDF | [`reports/ViolationIQ_IEEE_Report.pdf`](reports/ViolationIQ_IEEE_Report.pdf) |

The report explains the full ViolationIQ approach, architecture, datasets, model results, implemented modules, evidence-generation pipeline, and safety/manual-review design.

---

<!-- PROJECT_REPORT_END -->
'@

    if ($readme -match "<!-- PROJECT_REPORT_START -->") {
        $readme = [regex]::Replace(
            $readme,
            "(?s)<!-- PROJECT_REPORT_START -->.*?<!-- PROJECT_REPORT_END -->",
            $reportSection
        )
    } else {
        if ($readme -match "<!-- DEMO_VIDEO_END -->") {
            $readme = $readme -replace "<!-- DEMO_VIDEO_END -->", "<!-- DEMO_VIDEO_END -->`n$reportSection"
        } else {
            $readme = $readme + "`n" + $reportSection
        }
    }

    Set-Content -Path $mainReadme -Value $readme -Encoding UTF8
    Write-Host "Main README updated with project report links." -ForegroundColor Green
}

# 6. Push to GitHub
git add README.md reports/README.md reports/ViolationIQ_IEEE_Report.tex reports/ViolationIQ_IEEE_Report.pdf
git commit -m "Add IEEE project report source and PDF"
git push origin main
